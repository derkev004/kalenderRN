{%
\intcalcMod{%
\ifnum#2<3 #1-1\else#1\fi +%
\intcalcDiv{\ifnum#2<3 #1-1 \else#1\fi}{4} -%
\intcalcDiv{\ifnum#2<3 #1-1 \else#1\fi}{100} +%
\intcalcDiv{\ifnum#2<3 #1-1 \else#1\fi}{400} +%
\ifcase#2\or%
0\or%1
3\or%2
2\or%3
5\or%4
0\or%5
3\or%6
5\or%7
1\or%8
4\or%9
6\or%10
2\or%11
4\else%12
0\fi%
+ #3 + 6}{7}%
}
